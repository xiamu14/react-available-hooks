## Install

```bash
npm install --save react-available-hooks
```

## Usage

```tsx
import * as React from 'react'

import MyComponent from 'react-available-hooks'

class Example extends React.Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
